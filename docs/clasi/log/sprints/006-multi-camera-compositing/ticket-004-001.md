### Synthetic image generation

Generate images with ArUco 4x4 markers placed at known positions:
```python
import cv2
import numpy as np

def make_aruco_image(marker_positions, size=(640, 480)):
    """Create a synthetic image with ArUco 4x4 markers.
    marker_positions: dict of {marker_id: (center_x, center_y)}
    """
    img = np.zeros((size[1], size[0], 3), dtype=np.uint8) + 128  # gray background
    aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
    marker_size = 40
    for mid, (cx, cy) in marker_positions.items():
        marker_img = cv2.aruco.generateImageMarker(aruco_dict, mid, marker_size)
        marker_bgr = cv2.cvtColor(marker_img, cv2.COLOR_GRAY2BGR)
        x1 = cx - marker_size // 2
        y1 = cy - marker_size // 2
        img[y1:y1+marker_size, x1:x1+marker_size] = marker_bgr
    return img
```

Use the same 4 marker IDs (0-3) in both images but at different pixel positions (simulating different camera viewpoints).
