## Ticket 001: Composite class and cross-camera homography

Create `src/aprilcam/composite.py`:

```python
from __future__ import annotations
from dataclasses import dataclass, field
from typing import Optional
import uuid
import cv2 as cv
import numpy as np

@dataclass
class Composite:
    composite_id: str
    primary_camera_id: str
    secondary_camera_id: str
    homography: np.ndarray  # 3x3, maps secondary px -> primary px
    reprojection_error: float
    playfield_id: Optional[str] = None

class CompositeManager:
    def __init__(self):
        self._composites: dict[str, Composite] = {}

    def create(self, primary_id, secondary_id, homography, reprojection_error, playfield_id=None) -> Composite:
        cid = f"comp_{uuid.uuid4().hex[:8]}"
        c = Composite(cid, primary_id, secondary_id, homography, reprojection_error, playfield_id)
        self._composites[cid] = c
        return c

    def get(self, composite_id) -> Composite:
        return self._composites[composite_id]  # KeyError if missing

    def destroy(self, composite_id):
        del self._composites[composite_id]  # KeyError if missing

    def list(self) -> list[Composite]:
        return list(self._composites.values())


def compute_cross_camera_homography(primary_points, secondary_points):
    """Compute homography mapping secondary pixels to primary pixels.
    Returns (3x3 ndarray, rms_error). Raises ValueError if <4 points or degenerate."""
    pts_p = np.array(primary_points, dtype=np.float64).reshape(-1, 2)
    pts_s = np.array(secondary_points, dtype=np.float64).reshape(-1, 2)
    if len(pts_p) < 4 or len(pts_s) < 4:
        raise ValueError("At least 4 point correspondences required")
    H, mask = cv.findHomography(pts_s, pts_p, cv.RANSAC)
    if H is None:
        raise ValueError("Degenerate point configuration — could not compute homography")
    # Compute reprojection error
    pts_s_h = np.hstack([pts_s, np.ones((len(pts_s), 1))])
    projected = (H @ pts_s_h.T).T
    projected = projected[:, :2] / projected[:, 2:3]
    errors = np.sqrt(np.sum((projected - pts_p) ** 2, axis=1))
    rms = float(np.sqrt(np.mean(errors ** 2)))
    return H, rms


def map_tags_to_primary(detections, homography):
    """Transform tag detections from secondary to primary camera coordinates.
    detections: list of (pts_4x2_ndarray, raw, tag_id) tuples
    Returns list of dicts with id, center_px, corners_px, orientation_yaw."""
    import math
    if not detections:
        return []
    results = []
    for pts, _raw, tid in detections:
        pts_f = pts.astype(np.float64).reshape(-1, 1, 2)
        mapped = cv.perspectiveTransform(pts_f, homography).reshape(-1, 2)
        center = mapped.mean(axis=0)
        # Compute orientation from mapped corners
        top_mid = (mapped[0] + mapped[1]) / 2.0
        d = top_mid - center
        yaw = math.atan2(float(d[1]), float(d[0]))
        results.append({
            "id": int(tid),
            "center_px": [float(center[0]), float(center[1])],
            "corners_px": [[float(c[0]), float(c[1])] for c in mapped],
            "orientation_yaw": float(yaw),
        })
    return results
```
