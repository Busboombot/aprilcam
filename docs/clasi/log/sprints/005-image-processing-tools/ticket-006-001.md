### Setup

```python
import asyncio
import json
from pathlib import Path
import cv2
import numpy as np
import pytest

from aprilcam.mcp_server import (
    registry, playfield_registry, _motion_prev_frames,
    get_frame, crop_region, detect_lines, detect_circles,
    detect_contours, detect_motion, detect_qr_codes, apply_transform,
)

class FakeCapture:
    def __init__(self, img):
        self._img = img
    def read(self):
        return True, self._img.copy()
    def release(self):
        pass

@pytest.fixture(autouse=True)
def clean_registries():
    registry._cameras.clear()
    playfield_registry._playfields.clear()
    _motion_prev_frames.clear()
    yield
    registry._cameras.clear()
    playfield_registry._playfields.clear()
    _motion_prev_frames.clear()

@pytest.fixture
def camera_with_shapes():
    """Register a fake camera with a synthetic image containing shapes."""
    img = np.zeros((480, 640, 3), dtype=np.uint8)
    # Draw lines
    cv2.line(img, (100, 100), (500, 100), (255, 255, 255), 2)
    cv2.line(img, (100, 100), (100, 400), (255, 255, 255), 2)
    # Draw circle
    cv2.circle(img, (320, 240), 50, (255, 255, 255), 2)
    # Draw rectangle (for contour detection)
    cv2.rectangle(img, (400, 300), (550, 420), (255, 255, 255), -1)
    cam_id = registry.open(FakeCapture(img))
    return cam_id, img
```

### Tests

- `test_get_frame_base64` — call get_frame, verify returns ImageContent
- `test_get_frame_file` — call get_frame(format="file"), verify returns TextContent with path
- `test_get_frame_invalid_source` — unknown source, verify error
- `test_crop_region_success` — crop (100,100,200,200), verify returns image
- `test_crop_region_out_of_bounds` — crop entirely outside, verify error
- `test_detect_lines_finds_lines` — synthetic image with drawn lines, verify non-empty lines list
- `test_detect_lines_empty` — blank image, verify empty list
- `test_detect_circles_finds_circles` — synthetic image with circle, verify non-empty
- `test_detect_circles_empty` — blank image, verify empty
- `test_detect_contours_finds_shapes` — synthetic image with rectangle, verify contours with area and bbox
- `test_detect_motion_baseline` — first call returns is_baseline=true, empty regions
- `test_detect_motion_with_change` — register camera that returns different images on consecutive reads, verify non-empty regions
- `test_detect_qr_codes_empty` — no QR codes, verify empty list
- `test_apply_transform_canny` — apply canny transform, verify returns image
- `test_apply_transform_blur` — apply blur, verify returns image
- `test_apply_transform_unknown` — unknown operation, verify error

All tests use @pytest.mark.asyncio or _run() helper. Always parse JSON response: `data = json.loads(result[0].text)`.
