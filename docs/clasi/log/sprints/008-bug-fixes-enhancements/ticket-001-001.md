## Ticket: Unify taggen and arucogen into single taggen command with PDF output

### Requirements

Merge `arucogen` functionality into `taggen`. The unified `aprilcam taggen` command:

1. **Default output**: Multi-page PDF with one tag per page, page size 59mm x 102mm, tag centered with quiet zone.
2. **Default family**: AprilTag 36h11
3. **`-A` flag**: Switch to ArUco 4x4 family
4. **Positional arguments**: Comma-separated list of IDs or ranges. Examples:
   - `aprilcam taggen 0,1,2,3` — generates tags 0, 1, 2, 3
   - `aprilcam taggen 0-3` — same thing, range syntax
   - `aprilcam taggen 0-3,7,10-12` — mix of ranges and individual IDs
   - `aprilcam taggen -A 0-3` — ArUco 4x4 markers 0-3
5. **`--png` flag**: Output individual PNG files instead of PDF
6. **`--out` / `-o`**: Output file/directory path. For PDF: single file (default `tags.pdf`). For PNG: directory (default `tag-images/`).
7. **`--size`**: Tag image size in pixels (default 800)
8. **`--label`**: Add ID label below tag (for PNG mode)

def parse_ids(spec: str) -> list[int]:
    """Parse '0-3,7,10-12' into [0,1,2,3,7,10,11,12]"""
    ids = []
    for part in spec.split(','):
        part = part.strip()
        if '-' in part:
            start, end = part.split('-', 1)
            ids.extend(range(int(start), int(end) + 1))
        else:
            ids.append(int(part))
    return sorted(set(ids))

def render_tag(tag_id: int, family: str, size: int) -> np.ndarray:
    """Render a tag image for the given family."""
    if family == "aruco4x4":
        d = cv.aruco.getPredefinedDictionary(cv.aruco.DICT_4X4_50)
    else:
        d = cv.aruco.getPredefinedDictionary(cv.aruco.DICT_APRILTAG_36h11)
    img = cv.aruco.generateImageMarker(d, tag_id, size)
    return img  # grayscale

def generate_pdf(tag_ids, family, size, output_path, quiet_ratio=2.0/7.0):
    """Generate multi-page PDF, 59mm x 102mm pages, one tag per page."""
    from fpdf import FPDF
    import tempfile
    
    PAGE_W_MM = 59.0
    PAGE_H_MM = 102.0
    
    pdf = FPDF(unit="mm", format=(PAGE_W_MM, PAGE_H_MM))
    
    for tid in tag_ids:
        img = render_tag(tid, family, size)
        # Add quiet zone
        img_with_quiet = add_quiet_zone_gray(img, quiet_ratio)
        
        pdf.add_page()
        # Save tag to temp file
        tmp = tempfile.NamedTemporaryFile(suffix=".png", delete=False)
        cv.imwrite(tmp.name, img_with_quiet)
        
        # Center on page
        # Fit tag to page with margins
        margin = 3.0  # mm
        avail_w = PAGE_W_MM - 2 * margin
        avail_h = PAGE_H_MM - 2 * margin - 10  # leave room for label
        tag_size = min(avail_w, avail_h)
        x = (PAGE_W_MM - tag_size) / 2
        y = (PAGE_H_MM - tag_size) / 2 - 5  # shift up slightly for label
        pdf.image(tmp.name, x=x, y=y, w=tag_size, h=tag_size)
        
        # Add label
        family_label = "ArUco 4x4" if family == "aruco4x4" else "AprilTag 36h11"
        pdf.set_font("Helvetica", size=8)
        pdf.set_y(y + tag_size + 2)
        pdf.cell(PAGE_W_MM, 5, f"{family_label} ID: {tid}", align="C")
        
        # Clean up temp
        Path(tmp.name).unlink(missing_ok=True)
    
    pdf.output(str(output_path))

def add_quiet_zone_gray(img, quiet_ratio):
    h, w = img.shape[:2]
    pad = max(1, int(round(min(h, w) * float(quiet_ratio))))
    out = np.full((h + 2 * pad, w + 2 * pad), 255, dtype=np.uint8)
    out[pad:pad + h, pad:pad + w] = img
    return out
```

def main(argv=None):
    parser = argparse.ArgumentParser(description="Generate AprilTag or ArUco marker images")
    parser.add_argument("ids", nargs="?", default="0-3", help="Tag IDs: comma-separated list and/or ranges (e.g., '0-3,7,10-12')")
    parser.add_argument("-A", "--aruco", action="store_true", help="Generate ArUco 4x4 markers instead of AprilTag 36h11")
    parser.add_argument("--png", action="store_true", help="Output individual PNG files instead of PDF")
    parser.add_argument("-o", "--out", default=None, help="Output path (PDF file or PNG directory)")
    parser.add_argument("--size", type=int, default=800, help="Tag image size in pixels (default 800)")
    parser.add_argument("--label", action="store_true", help="Add ID label below tag (PNG mode only)")
    args = parser.parse_args(argv)
    
    family = "aruco4x4" if args.aruco else "36h11"
    tag_ids = parse_ids(args.ids)
    
    if args.png:
        # PNG mode - output individual files
        out_dir = Path(args.out or "tag-images")
        out_dir.mkdir(parents=True, exist_ok=True)
        for tid in tag_ids:
            img = render_tag(tid, family, args.size)
            img_bgr = cv.cvtColor(img, cv.COLOR_GRAY2BGR)
            img_bgr = add_quiet_zone(img_bgr, 2.0/7.0)
            if args.label:
                label = f"{'ArUco 4x4' if family == 'aruco4x4' else 'AprilTag 36h11'} ID: {tid}"
                img_bgr = add_label_below(img_bgr, label)
            prefix = "aruco4x4" if family == "aruco4x4" else "tag36h11"
            cv.imwrite(str(out_dir / f"{prefix}_{tid}.png"), img_bgr)
        print(f"Wrote {len(tag_ids)} images to {out_dir.resolve()}")
    else:
        # PDF mode (default)
        out_path = Path(args.out or "tags.pdf")
        generate_pdf(tag_ids, family, args.size, out_path)
        print(f"Wrote {len(tag_ids)} tags to {out_path.resolve()}")
    return 0
```
